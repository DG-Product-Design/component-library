#### We use the following breakpoints across all our applications:
- [XXS] : .text-headline-xxs
- [XS] : .text-headline-xs
- [SM] : .text-body-sm, .text-headline-sm
- [SM + BOLD] : .text-body-sm-bold, .text-headline-sm-bold
- [MD] : .text-body-md, .text-headline-md
- [MD + BOLD] : .text-body-sm-bold, .text-headline-sm-bold
- [LG] : .text-body-lg, .text-headline-lg
- [LG + BOLD] : .text-body-lg-bold, .text-headline-lg-bold
- [XL] : .text-headline-xl
- [2XL] : .text-headline-2xl
- [3XL] : .text-headline-3xl
- [4XL] : .text-headline-4xl
- [CAPTION] : .text-caption
